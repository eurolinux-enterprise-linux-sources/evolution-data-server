struct CamelHttpStream

struct CamelHttpStream {
	CamelStream parent_object;

	CamelMimeParser *parser;

	CamelContentType *content_type;
	struct _camel_header_raw *headers;

	CamelHttpMethod method;
	struct _CamelSession *session;
	CamelURL *url;

	gchar *user_agent;

	/* proxy info */
	CamelURL *proxy;
	gchar *authrealm;
	gchar *authpass;

	gint statuscode;

	CamelStream *raw;
	CamelStream *read;
};
